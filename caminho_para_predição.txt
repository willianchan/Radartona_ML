SELECT 
      [origem] as id_radar
      ,[v0] as velocidade
	  ,YEAR(data_inicio) as ano
	  ,MONTH(data_inicio) as mes
	  ,DAY(data_inicio) as dia
	  ,datepart(hour, data_inicio) as hora
	  ,datepart(minute, data_inicio) as minuto
  FROM [radartona].[dbo].[trajetos]
